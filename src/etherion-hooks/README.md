## Install

```bash
npm install --save etherion-hooks
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'etherion-hooks'
import 'etherion-hooks/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
